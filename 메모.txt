CREATE TABLE `Schedule` (
	`schedule_idx`	number	NOT NULL,
	`yy`	number	NOT NULL,
	`mm`	number	NOT NULL,
	`dd`	number	NOT NULL,
	`emp_idx`	number	NOT NULL
);

CREATE TABLE `schedule_detail` (
	`title`	varchar2(20)	NULL,
	`detail`	varchar2(100)	NULL,
	`filename`	varchar2(30)	NULL,
	`imgname`	varchar2(30)	NULL,
	`schedule_idx`	number	NOT NULL
);

ALTER TABLE `Schedule` ADD CONSTRAINT `PK_SCHEDULE` PRIMARY KEY (
	`schedule_idx`
);

	`emp_idx`

	`emp_idx`
